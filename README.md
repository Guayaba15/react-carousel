## Install

```bash
npm install --save react-carousel-project
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-carousel-project'
import 'react-carousel-project/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
